public class PassByValueOrReference
{
   static void addoneV(int a, float b)
   {
      a++;
      b += 1;
   }
   
   static void changeStringV(String x)
   {
      x += " add some";
   }
   
   static void addoneR(int [] a, float [] b)
   {
      a[0] ++;
      b[0] += 1;
   }
   
   static void changeStringR(String [] x)
   {
      x[0] += " add some";
   }
   
   static class AInner
   {
      int right=0;
      double left =2;
   }
   static void addoneC(AInner w)
   {
      w.right ++;
      w.left += 1;
   }
   
   public static void main(String [] args)
   {
      int a = 1; float b = 1;
      System.out.println("Before calling addoneV, a="+ a +", b="+ b);
      addoneV(a, b);
      System.out.println("After   calling addoneV, a="+ a +", b="+ b);
      
      int [] aa = new int[]
      {
         a
      };
      float [] ba = new float[]
      {
         b
      };
      
      System.out.println("Before calling addoneV, aa[0]="+ aa[0]);
      addoneR(aa, ba);
      System.out.println("After   calling addoneV, ba[0]="+ ba[0]);
      
      String original = "We are students";
      
      System.out.println("Before calling changeStringV, original="+ original);
      changeStringV(original);
      System.out.println("After   calling changeStringV, original="+ original);
      
      String [] originala = new String[]
      {
         original
      };
      
      System.out.println("Before calling changeStringR, original="+ original);
      changeStringR(originala);
      System.out.println("After   calling changeStringV, originala[0]="+ originala[0]);
      
      AInner y = new AInner();
      System.out.println("Before calling addoneC, y.left="+ ", y.right="+ y.right);
      addoneC(y);
      System.out.println("After  calling addoneC, y.left="+ ", y.right="+ y.right);
   }
   
}

import java.util.function.*;
public class Main
{
   static void bubbleSort(int arr[], BiFunction compare)
   {
      int n = arr.length;
      for (int i = 0; i < n - 1; i++)
         for (int j = 0; j < n - i - 1; j++)
         {
            Object dif = (compare.apply(arr[j],arr[j + 1]));
            if ( ((Integer)dif).intValue() > 0)
            {
               int temp = arr[j];
               arr[j] = arr[j + 1];
               arr[j + 1] = temp;
            }
         }
   }
   public static void main(String[] args)
   {
      int []x =
      {
         45,34,2,5,1,564,3
      };
      BiFunction compare = (a, b) -> a-b;
      bubbleSort(x, compare);
      for (int i=0; i < x.length; i++)
         System.out.println(x[i] + " ");
   }
}
